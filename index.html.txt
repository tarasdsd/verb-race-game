<script>
    const gameData = [
        { s: "I visited my grandmother last Sunday.", v: "visited" },
        { s: "She forgot her umbrella in the car.", v: "forgot" },
        { s: "We watched a funny movie together.", v: "watched" },
        { s: "The cat drank all the warm milk.", v: "drank" },
        { s: "He bought a red balloon at the fair.", v: "bought" },
        { s: "They ran very fast to catch the bus.", v: "ran" }
    ];

    let current = 0;
    let score = 0;

    const micBtn = document.getElementById('micBtn');
    const retryBtn = document.getElementById('retry');
    const sentenceDiv = document.getElementById('sentence');
    const feedback = document.getElementById('feedback');
    const car = document.getElementById('car');
    const road = document.getElementById('road');

    // Speech API setup
    const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
    const recog = new SpeechRecognition();
    recog.lang = 'en-US';

    function loadQuestion() {
        sentenceDiv.innerText = gameData[current].s;
        micBtn.style.display = "inline-block";
        retryBtn.style.display = "none";
        car.classList.remove('crash');
        road.classList.remove('scrolling');
        feedback.innerText = "Get ready...";
    }

    micBtn.onclick = () => {
        recog.start();
        feedback.innerText = "Listening... Speak now!";
        micBtn.disabled = true;
    };

    recog.onresult = (event) => {
        const result = event.results[0][0].transcript.toLowerCase().replace(/[.,!]/g, "");
        const target = gameData[current].s.toLowerCase().replace(/[.,!]/g, "");

        if (result === target) {
            winRound();
        } else {
            failRound(result);
        }
    };

    function winRound() {
        feedback.innerText = "Great job! Keep going!";
        road.classList.add('scrolling');
        score += 10;
        document.getElementById('score').innerText = score;
        
        confetti({ particleCount: 100, spread: 70, origin: { y: 0.6 } });

        setTimeout(() => {
            current++;
            if(current < gameData.length) {
                loadQuestion();
                micBtn.disabled = false;
            } else {
                sentenceDiv.innerText = "🏆 CHAMPION! You mastered the verbs!";
                feedback.innerText = "Game Over - Well done!";
            }
        }, 3000);
    }

    function failRound(said) {
        feedback.innerText = `The car crashed! You said: "${said}"`;
        car.classList.add('crash');
        micBtn.style.display = "none";
        retryBtn.style.display = "inline-block";
        micBtn.disabled = false;
    }

    function resetGame() {
        loadQuestion();
    }

    window.onload = loadQuestion;
</script>
